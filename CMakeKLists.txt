cmake_minimum_required(VERSION 3.22)
project(IncrediBoxer VERSION 0.1.0)

set(CMAKE_CXX_STANDARD 20)

# JUCE and SDK
add_subdirectory(extern/JUCE)
set(VST3_SDK_DIR "C:/SDKs/vst3sdk" CACHE PATH "VST3 SDK Path")

juce_add_plugin(IncrediBoxer
    COMPANY_NAME "YourName"
    PRODUCT_NAME "IncrediBoxer"
    IS_SYNTH TRUE
    NEEDS_MIDI_INPUT TRUE
    FORMATS VST3
)

target_sources(IncrediBoxer PRIVATE
    Source/PluginProcessor.cpp
    Source/PluginEditor.cpp)

target_link_libraries(IncrediBoxer PRIVATE
    juce::juce_audio_utils
    juce::juce_dsp
    juce::juce_gui_extra)
